\subsection{Mục tiêu - Phạm vi dự án}
\subsubsection{Mục tiêu}
\begin{itemize}[label = -]
    \item \textbf{Kiểm soát ra vào:} Tự động hóa quá trình kiểm tra phương tiện và xử lý linh hoạt các trường hợp đặc biệt.
    \vspace{-10pt}
    \item \textbf{Quản lý vị trí:} Theo dõi và cập nhật trạng thái các vị trí đỗ xe theo thời gian thực.
    \vspace{-10pt}
    \item \textbf{Thanh toán tự động:} Tích hợp các cổng thanh toán điện tử để tối ưu hóa quy trình thu phí.
    \vspace{-10pt}
    \item \textbf{Đảm bảo tính ổn định:} Hệ thống đạt độ tin cậy cao khi xử lý lưu lượng lớn và có khả năng hoạt động cục bộ khi mất kết nối mạng.
    \vspace{-10pt}
    \item \textbf{Xác thực chính xác:} Đảm bảo tính nhất quán giữa thẻ xe và phương tiện (đúng thẻ - đúng xe).
\end{itemize}
\subsubsection{Phạm vi dự án}
\begin{itemize}[label=-]
    \item \textbf{Đối tượng phục vụ:} Bao gồm thành viên nội bộ (Cán bộ, Sinh viên) và khách vãng lai.
    \vspace{-10pt}
    \item \textbf{Hệ thống thông tin:} Kết nối với HCMUT\_DATACORE (đồng bộ dữ liệu), HCMUT\_SSO (định danh) và BKPay (thanh toán).
    \vspace{-10pt}
    \item \textbf{Hạ tầng phần cứng:} Triển khai các thiết bị cảm biến (sensors), rào chắn (barrier), màn hình hiển thị, còi báo động (sirens) và khối điều khiển trung tâm (MCU).
    \vspace{-10pt}
    \item \textbf{Giới hạn thực hiện:} Tập trung phát triển sản phẩm khả thi tối thiểu (MVP), không triển khai toàn diện hệ thống Back-end, do đó hạn chế xử lý các luồng ngoại lệ phức tạp (exception flow).
\end{itemize}
